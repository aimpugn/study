/**
 * - if 야근, then 피로도 쌓임
 * - 피로도 = 야근 시작 시점에서 남은 일의 작업량을 제곱하여 더한 값
 * - 1시간 동안 작업량 1 처리 가능

 * ex:
 * - works: [4, 3, 3]
 *   n: 4
 *   => [4 -2, 3 -1, 3 - 1]
 *   => [2, 2, 2]
 *   => 4 + 4 + 4 = 12
 *
 * - works: [2, 1, 2]
 *   n: 1
 *   => [2 - 1, 1, 2]
 *   => 1 + 1 + 4 = 6
 *
 * @param n 퇴근까지 남은 시간
 *          1 <= n <= 1,000,000
 * @param works 각 일에 대한 작업량
 *              1 <= works.length <= 20,000
 *              1 <= work <= 50,000
 * @return 야근 피로도를 최소화한 값
 */
public long solution(int n, int[] works) {
    long answer = 0;

    // n    : 4

    // - [0, 3, 3] = 18

    //   -4
    // - [1, 2, 3] = 14
    //    ^  ^
    //   -3 -1
    // - [1, 3, 2] = 14 // [1, 2, 3]과 같지만 결과는 같습니다. memoization
    //    ^  ^  ^
    //   -3  0 -1

    //    ^  ^
